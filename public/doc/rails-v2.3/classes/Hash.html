def deep_symbolize_keys
  inject({}) { |result, (key, value)|
    value = value.deep_symbolize_keys if value.is_a?(Hash)
    result[(key.to_sym rescue key) || key] = value
    result
  }
end

def slice(*keep_keys)
  h = {}
  keep_keys.each { |key| h[key] = fetch(key) }
  h
end
