Ensures that the passed record is valid by Active Record standards and returns any error messages if it is not.

def assert_valid(record)
  ::ActiveSupport::Deprecation.warn("assert_valid is deprecated. Use assert record.valid? instead", caller)
  clean_backtrace do
    assert record.valid?, record.errors.full_messages.join("\n")
  end
end
