def backtrace_cleaner
  @@backtrace_cleaner ||= begin
    # Relies on ActiveSupport, so we have to lazy load to postpone definition until AS has been loaded
    require 'rails/backtrace_cleaner'
    Rails::BacktraceCleaner.new
  end
end
