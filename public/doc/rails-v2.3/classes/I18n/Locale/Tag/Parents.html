def parent
  @parent ||= begin
    segs = to_a.compact
    segs.length > 1 ? self.class.tag(*segs[0..(segs.length-2)].join('-')) : nil
  end
end

def parents
  @parents ||= ([parent] + (parent ? parent.parents : [])).compact
end
